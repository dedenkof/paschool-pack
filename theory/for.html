                        let mainBlock = document.querySelector('.main-block');

                        // Выводим строки
                        for (let i = 1; i <= 8; i++){

                            row = document.createElement('div');

                            mainBlock.appendChild(row);

                            // Выводим колонки в строке
                            for (let j = 1; j <= 8; j++){

                                col = document.createElement('div');

                                mainBlock.appendChild(col);

                                if((i + j) % 2 == 0){

                                    col.className = 'square white';

                                }else{

                                    col.className = 'square black';

                                }

                            }
                        }

                    function chessBoardRender(){
                        let mainBlock = document.querySelector('.main-block');
                        let square;
                        let flag = true;

                        for (let i = 0; i < 8; i++){

                            for (let j = 0; j < 8; j++){

                                if(j == 0) flag = !flag;

                                square = document.createElement('div');

                                if(flag){

                                    square.className = 'square black';

                                }else{

                                    square.className = 'square white';

                                }

                                mainBlock.appendChild(square);

                                flag = !flag;

                            }
                        }
                    }

                    chessBoardRender();
